    rows =  input('엑셀파일에서 서로 곱하고자 하는 열을 순서대로 입력해주세요.(각 숫자는 띄어쓰기 해주세요)') #유효숫자를 처리할 열을 입력받는다.  
    r=rows.split()  
    answer=[]  
    for row in data:  
        l=[]  
        for i in range(len(r)):  
            l.append(row[r[i]])  
        answer.append(multforcsv(l))  
    writer= csv.writer(newfile)  
    writer.writerow(answer)
