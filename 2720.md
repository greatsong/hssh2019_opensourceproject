    cnt = 0  
    # 소수로만 되어있는 경우  
    if a == 0:  
        b = a  
        while b != 0.0 :    
            cnt += 1  
            b = b*10  
            #소수부분 구하기  
            l = len(str(int(b)))  
            b = float('0'+str(b)[l:])  
    # 정수부분이 있는 경우  
    b = a  
    while b == int(b):  
        b = b/10  
        if int(b):  
            cnt-=1  
    return cnt
